def str_int(number):
    return str(int(number))


def str_float(number, decimals=8, force=False):
    if force:
        return ("{0:."+str(decimals)+"f}").format(number)
    elif number % 1:
        return str(round(number, decimals))
    else:
        return str_int(number)
